144  inline void initialize() {
145  if( !my_initialized ) {
146  internal_initialize();
147 #if TBB_USE_THREADING_TOOLS
148  // Threading tools respect lock prefix but report false-positive data-race via plain store
149  internal::as_atomic(my_initialized).fetch_and_store<release>(true);
150 #else
151  my_initialized = true;
152 #endif //TBB_USE_THREADING_TOOLS
153  }
154  }
155 
157  inline void initialize(int max_concurrency, unsigned reserved_for_masters = 1) {
158  __TBB_ASSERT( !my_arena, "Impossible to modify settings of an already initialized task_arena");
159  if( !my_initialized ) {
160  my_max_concurrency = max_concurrency;
161  my_master_slots = reserved_for_masters;
162  initialize();
163  }
164  }
165 
168  inline void terminate() {
169  if( my_initialized ) {
170  internal_terminate();
171  my_initialized = false;
172  }
173  }
174 
177  ~task_arena() {
178  terminate();
179  }
180 
183  bool is_active() const { return my_initialized; }
