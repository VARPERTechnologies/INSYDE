473 {

476  INPUT: pointer to the tour that undergoes local optimization
477  OUTPUT: none
478  (SIDE)EFFECTS: tour is 2-opt
479  COMMENTS: the neighbourhood is scanned in random order (this need
480  not be the best possible choice). Concerning the speed-ups used
481  here consult, for example, Chapter 8 of
482  Holger H. Hoos and Thomas Stuetzle,
483  Stochastic Local Search---Foundations and Applications,
484  Morgan Kaufmann Publishers, 2004.
485  or some of the papers online available from David S. Johnson.
486  */
487  long int c1, c2; /* cities considered for an exchange */
488  long int s_c1, s_c2; /* successor cities of c1 and c2 */
489  long int p_c1, p_c2; /* predecessor cities of c1 and c2 */
490  long int pos_c1, pos_c2; /* positions of cities c1, c2 */
491  long int i, j, h, l;
492  long int improvement_flag, improve_node, help, n_improves = 0, n_exchanges=0;
493  long int h1=0, h2=0, h3=0, h4=0;
494  long int radius; /* radius of nn-search */
495  long int gain = 0;
496  long int *random_vector;
497  long int *pos; /* positions of cities in tour */
498  long int *dlb; /* vector containing don't look bits */
499 
500  pos = (long int*) malloc(n * sizeof(long int));
501  dlb = (long int*) malloc(n * sizeof(long int));
502  for ( i = 0 ; i < n ; i++ ) {
503  pos[tour[i]] = i;
504  dlb[i] = 'f';
505  }
506 
507  improvement_flag = 't';
508  random_vector = generate_random_permutation( n );
509 
510  while ( improvement_flag ) {
511 
512  improvement_flag = 'f';
513 
514  for (l = 0 ; l < n; l++) {
515 
516  c1 = random_vector[l];
517  DEBUG ( assert ( c1 < n && c1 >= 0); )
518  if ( dlb_flag && dlb[c1] )
519  continue;
520  improve_node = 'f';
521  pos_c1 = pos[c1];
522  s_c1 = tour[pos_c1+1];
523  radius = instance.distance[c1][s_c1];
524 
525  /* First search for c1's nearest neighbours, use successor of c1 */
526  for ( h = 0 ; h < nn_ls ; h++ ) {
527  c2 = instance.nn_list[c1][h]; /* exchange partner, determine its position */
528  if ( radius > instance.distance[c1][c2] ) {
529  s_c2 = tour[pos[c2]+1];
530  gain = - radius + instance.distance[c1][c2] +
531  instance.distance[s_c1][s_c2] - instance.distance[c2][s_c2];
532  if ( gain < 0 ) {
533  h1 = c1; h2 = s_c1; h3 = c2; h4 = s_c2;
534  improve_node = 't';
535  goto exchange2opt;
536  }
537  }
538  else
539  break;
540  }
541  /* Search one for next c1's h-nearest neighbours, use predecessor c1 */
542  if (pos_c1 > 0)
543  p_c1 = tour[pos_c1-1];
544  else
545  p_c1 = tour[n-1];
546  radius = instance.distance[p_c1][c1];
547  for ( h = 0 ; h < nn_ls ; h++ ) {
548  c2 = instance.nn_list[c1][h]; /* exchange partner, determine its position */
549  if ( radius > instance.distance[c1][c2] ) {
550  pos_c2 = pos[c2];
551  if (pos_c2 > 0)
552  p_c2 = tour[pos_c2-1];
553  else
554  p_c2 = tour[n-1];
555  if ( p_c2 == c1 )
556  continue;
557  if ( p_c1 == c2 )
558  continue;
559  gain = - radius + instance.distance[c1][c2] +
560  instance.distance[p_c1][p_c2] - instance.distance[p_c2][c2];
561  if ( gain < 0 ) {
562  h1 = p_c1; h2 = c1; h3 = p_c2; h4 = c2;
563  improve_node = 't';
564  goto exchange2opt;
565  }
566  }
567  else
568  break;
569  }
570  if (improve_node) {
571  exchange2opt:
572  n_exchanges++;
573  improvement_flag = 't';
574  dlb[h1] = 'f'; dlb[h2] = 'f';
575  dlb[h3] = 'f'; dlb[h4] = 'f';
576  /* Now perform move */
577  if ( pos[h3] < pos[h1] ) {
578  help = h1; h1 = h3; h3 = help;
579  help = h2; h2 = h4; h4 = help;
580  }
581  if ( pos[h3] - pos[h2] < n / 2 + 1) {
582  /* reverse inner part from pos[h2] to pos[h3] */
583  i = pos[h2]; j = pos[h3];
584  while (i < j) {
585  c1 = tour[i];
586  c2 = tour[j];
587  tour[i] = c2;
588  tour[j] = c1;
589  pos[c1] = j;
590  pos[c2] = i;
591  i++; j--;
592  }
593  }
594  else {
595  /* reverse outer part from pos[h4] to pos[h1] */
596  i = pos[h1]; j = pos[h4];
597  if ( j > i )
598  help = n - (j - i) + 1;
599  else
600  help = (i - j) + 1;
601  help = help / 2;
602  for ( h = 0 ; h < help ; h++ ) {
603  c1 = tour[i];
604  c2 = tour[j];
605  tour[i] = c2;
606  tour[j] = c1;
607  pos[c1] = j;
608  pos[c2] = i;
609  i--; j++;
610  if ( i < 0 )
611  i = n-1;
612  if ( j >= n )
613  j = 0;
614  }
615  tour[n] = tour[0];
616  }
617  } else {
618  dlb[c1] = 't';
619  }
620 
621  }
622  if ( improvement_flag ) {
623  n_improves++;
624  }
625  }

627  free( dlb );
628  free( pos );
