61  aggregator_generic() : handler_busy(false) { pending_operations = NULL; }

71  template < typename handler_type >
72  void execute(operation_type *op, handler_type &handle_operations, bool long_life_time = true) {
73  operation_type *res;
74  // op->status should be read before inserting the operation in the
75  // aggregator queue since it can become invalid after executing a
76  // handler (if the operation has 'short' life time.)
77  const uintptr_t status = op->status;
78 
79  // ITT note: &(op->status) tag is used to cover accesses to this op node. This
80  // thread has created the operation, and now releases it so that the handler
81  // thread may handle the associated operation w/o triggering a race condition;
82  // thus this tag will be acquired just before the operation is handled in the
83  // handle_operations functor.
84  call_itt_notify(releasing, &(op->status));
85  // insert the operation in the queue.
86  do {
87  // ITT may flag the following line as a race; it is a false positive:
88  // This is an atomic read; we don't provide itt_hide_load_word for atomics
89  op->next = res = pending_operations; // NOT A RACE
90  } while (pending_operations.compare_and_swap(op, res) != res);
91  if (!res) { // first in the list; handle the operations.
92  // ITT note: &pending_operations tag covers access to the handler_busy flag,
93  // which this waiting handler thread will try to set before entering
94  // handle_operations.
95  call_itt_notify(acquired, &pending_operations);
96  start_handle_operations(handle_operations);
97  // The operation with 'short' life time can already be destroyed.
98  if (long_life_time)
99  __TBB_ASSERT(op->status, NULL);
100  }
101  // not first; wait for op to be ready.
102  else if (!status) { // operation is blocking here.
103  __TBB_ASSERT(long_life_time, "The blocking operation cannot have 'short' life time. Since it can already be destroyed.");
104  call_itt_notify(prepare, &(op->status));
105  spin_wait_while_eq(op->status, uintptr_t(0));
106  itt_load_word_with_acquire(op->status);
107  }
108  }
109 
110  private:
112  atomic<operation_type *> pending_operations;
114  uintptr_t handler_busy;
115 
117  template < typename handler_type >
118  void start_handle_operations( handler_type &handle_operations ) {
119  operation_type *op_list;
120 
121  // ITT note: &handler_busy tag covers access to pending_operations as it is passed
122  // between active and waiting handlers. Below, the waiting handler waits until
123  // the active handler releases, and the waiting handler acquires &handler_busy as
124  // it becomes the active_handler. The release point is at the end of this
125  // function, when all operations in pending_operations have been handled by the
126  // owner of this aggregator.
127  call_itt_notify(prepare, &handler_busy);
128  // get the handler_busy:
129  // only one thread can possibly spin here at a time
130  spin_wait_until_eq(handler_busy, uintptr_t(0));
131  call_itt_notify(acquired, &handler_busy);
132  // acquire fence not necessary here due to causality rule and surrounding atomics
133  __TBB_store_with_release(handler_busy, uintptr_t(1));
134 
135  // ITT note: &pending_operations tag covers access to the handler_busy flag
136  // itself. Capturing the state of the pending_operations signifies that
137  // handler_busy has been set and a new active handler will now process that list's
138  // operations.
139  call_itt_notify(releasing, &pending_operations);
140  // grab pending_operations
141  op_list = pending_operations.fetch_and_store(NULL);
142 
143  // handle all the operations
144  handle_operations(op_list);
145 
146  // release the handler
147  itt_store_word_with_release(handler_busy, uintptr_t(0));
148  }
149 };

151 template < typename handler_type, typename operation_type >
152 class aggregator : public aggregator_generic<operation_type> {
153  handler_type handle_operations;
154 public:
155  aggregator() {}
156  explicit aggregator(handler_type h) : handle_operations(h) {}
157 
158  void initialize_handler(handler_type h) { handle_operations = h; }
159 
160  void execute(operation_type *op) {
161  aggregator_generic<operation_type>::execute(op, handle_operations);
162  }
