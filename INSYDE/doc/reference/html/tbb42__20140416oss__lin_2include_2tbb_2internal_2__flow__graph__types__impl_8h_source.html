172 // type which contains another type, tests for what type is contained, and references to it.
173 // internal::Wrapper<T>
174 // void CopyTo( void *newSpace) : builds a Wrapper<T> copy of itself in newSpace
175 
176 // struct to allow us to copy and test the type of objects
177 struct WrapperBase {
178  virtual ~WrapperBase() {}
179  virtual void CopyTo(void* /*newSpace*/) const { }
180 };
181 
182 // Wrapper<T> contains a T, with the ability to test what T is. The Wrapper<T> can be
183 // constructed from a T, can be copy-constructed from another Wrapper<T>, and can be
184 // examined via value(), but not modified.
185 template<typename T>
186 struct Wrapper: public WrapperBase {
187  typedef T value_type;
188  typedef T* pointer_type;
189 private:
190  T value_space;
191 public:
192  const value_type &value() const { return value_space; }
193 
194 private:
195  Wrapper();
196 
197  // on exception will ensure the Wrapper will contain only a trivially-constructed object
198  struct _unwind_space {
199  pointer_type space;
200  _unwind_space(pointer_type p) : space(p) {}
201  ~_unwind_space() {
202  if(space) (void) new (space) Wrapper<default_constructed>(default_constructed());
203  }
204  };
205 public:
206  explicit Wrapper( const T& other ) : value_space(other) { }
207  explicit Wrapper(const Wrapper& other) : value_space(other.value_space) { }
208 
209  /*override*/void CopyTo(void* newSpace) const {
210  _unwind_space guard((pointer_type)newSpace);
211  (void) new(newSpace) Wrapper(value_space);
212  guard.space = NULL;
213  }
214  /*override*/~Wrapper() { }
215 };

217 // specialization for array objects
218 template<typename T, size_t N>
219 struct Wrapper<T[N]> : public WrapperBase {
220  typedef T value_type;
221  typedef T* pointer_type;
222  // space must be untyped.
223  typedef T ArrayType[N];
224 private:
225  // The space is not of type T[N] because when copy-constructing, it would be
226  // default-initialized and then copied to in some fashion, resulting in two
227  // constructions and one destruction per element. If the type is char[ ], we
228  // placement new into each element, resulting in one construction per element.
229  static const size_t space_size = sizeof(ArrayType) / sizeof(char);
230  char value_space[space_size];
231 
232 
233  // on exception will ensure the already-built objects will be destructed
234  // (the value_space is a char array, so it is already trivially-destructible.)
235  struct _unwind_class {
236  pointer_type space;
237  int already_built;
238  _unwind_class(pointer_type p) : space(p), already_built(0) {}
239  ~_unwind_class() {
240  if(space) {
241  for(size_t i = already_built; i > 0 ; --i ) space[i-1].~value_type();
242  (void) new(space) Wrapper<default_constructed>(default_constructed());
243  }
244  }
245  };
246 public:
247  const ArrayType &value() const {
248  char *vp = const_cast<char *>(value_space);
249  return reinterpret_cast<ArrayType &>(*vp);
250  }
251 
252 private:
253  Wrapper();
254 public:
255  // have to explicitly construct because other decays to a const value_type*
256  explicit Wrapper(const ArrayType& other) {
257  _unwind_class guard((pointer_type)value_space);
258  pointer_type vp = reinterpret_cast<pointer_type>(&value_space);
259  for(size_t i = 0; i < N; ++i ) {
260  (void) new(vp++) value_type(other[i]);
261  ++(guard.already_built);
262  }
263  guard.space = NULL;
264  }
265  explicit Wrapper(const Wrapper& other) : WrapperBase() {
266  // we have to do the heavy lifting to copy contents
267  _unwind_class guard((pointer_type)value_space);
268  pointer_type dp = reinterpret_cast<pointer_type>(value_space);
269  pointer_type sp = reinterpret_cast<pointer_type>(const_cast<char *>(other.value_space));
270  for(size_t i = 0; i < N; ++i, ++dp, ++sp) {
271  (void) new(dp) value_type(*sp);
272  ++(guard.already_built);
273  }
274  guard.space = NULL;
275  }
276 
277  /*override*/void CopyTo(void* newSpace) const {
278  (void) new(newSpace) Wrapper(*this); // exceptions handled in copy constructor
279  }
280 
281  /*override*/~Wrapper() {
282  // have to destroy explicitly in reverse order
283  pointer_type vp = reinterpret_cast<pointer_type>(&value_space);
284  for(size_t i = N; i > 0 ; --i ) vp[i-1].~value_type();
285  }
286 };

408  // assignment must destroy and re-create the Wrapper type, as there is no way
409  // to create a Wrapper-to-Wrapper assign even if we find they agree in type.

412  WrapperBase *h = punned_cast<WrapperBase *>(&my_space);
413  h->~WrapperBase();
414  const WrapperBase *ch = punned_cast<const WrapperBase *>(&(rhs.my_space));

416  }
