264  void handle_operations(cpq_operation *op_list) {
265  cpq_operation *tmp, *pop_list=NULL;
266 
267  __TBB_ASSERT(mark == data.size(), NULL);
268 
269  // First pass processes all constant (amortized; reallocation may happen) time pushes and pops.
270  while (op_list) {
271  // ITT note: &(op_list->status) tag is used to cover accesses to op_list
272  // node. This thread is going to handle the operation, and so will acquire it
273  // and perform the associated operation w/o triggering a race condition; the
274  // thread that created the operation is waiting on the status field, so when
275  // this thread is done with the operation, it will perform a
276  // store_with_release to give control back to the waiting thread in
277  // aggregator::insert_operation.
278  call_itt_notify(acquired, &(op_list->status));
279  __TBB_ASSERT(op_list->type != INVALID_OP, NULL);
280  tmp = op_list;
281  op_list = itt_hide_load_word(op_list->next);
282  if (tmp->type == PUSH_OP) {
283  __TBB_TRY {
284  data.push_back(*(tmp->elem));
285  __TBB_store_with_release(my_size, my_size+1);
286  itt_store_word_with_release(tmp->status, uintptr_t(SUCCEEDED));
287  } __TBB_CATCH(...) {
288  itt_store_word_with_release(tmp->status, uintptr_t(FAILED));
289  }
290  }
291  else { // tmp->type == POP_OP
292  __TBB_ASSERT(tmp->type == POP_OP, NULL);
293  if (mark < data.size() &&
294  compare(data[0], data[data.size()-1])) {
295  // there are newly pushed elems and the last one
296  // is higher than top
297  *(tmp->elem) = data[data.size()-1]; // copy the data
298  __TBB_store_with_release(my_size, my_size-1);
299  itt_store_word_with_release(tmp->status, uintptr_t(SUCCEEDED));
300  data.pop_back();
301  __TBB_ASSERT(mark<=data.size(), NULL);
302  }
303  else { // no convenient item to pop; postpone
304  itt_hide_store_word(tmp->next, pop_list);
305  pop_list = tmp;
306  }
307  }
308  }
309 
310  // second pass processes pop operations
311  while (pop_list) {
312  tmp = pop_list;
313  pop_list = itt_hide_load_word(pop_list->next);
314  __TBB_ASSERT(tmp->type == POP_OP, NULL);
315  if (data.empty()) {
316  itt_store_word_with_release(tmp->status, uintptr_t(FAILED));
317  }
318  else {
319  __TBB_ASSERT(mark<=data.size(), NULL);
320  if (mark < data.size() &&
321  compare(data[0], data[data.size()-1])) {
322  // there are newly pushed elems and the last one is
323  // higher than top
324  *(tmp->elem) = data[data.size()-1]; // copy the data
325  __TBB_store_with_release(my_size, my_size-1);
326  itt_store_word_with_release(tmp->status, uintptr_t(SUCCEEDED));
327  data.pop_back();
328  }
329  else { // extract top and push last element down heap
330  *(tmp->elem) = data[0]; // copy the data
331  __TBB_store_with_release(my_size, my_size-1);
332  itt_store_word_with_release(tmp->status, uintptr_t(SUCCEEDED));
333  reheap();
334  }
335  }
336  }
337 
338  // heapify any leftover pushed elements before doing the next
339  // batch of operations
340  if (mark<data.size()) heapify();
341  __TBB_ASSERT(mark == data.size(), NULL);
342  }
343 
345  void heapify() {
346  if (!mark && data.size()>0) mark = 1;
347  for (; mark<data.size(); ++mark) {
348  // for each unheapified element under size
349  size_type cur_pos = mark;
350  value_type to_place = data[mark];
351  do { // push to_place up the heap
352  size_type parent = (cur_pos-1)>>1;
353  if (!compare(data[parent], to_place)) break;
354  data[cur_pos] = data[parent];
355  cur_pos = parent;
356  } while( cur_pos );
357  data[cur_pos] = to_place;
358  }
359  }
360 
362 
363  void reheap() {
364  size_type cur_pos=0, child=1;
365 
366  while (child < mark) {
367  size_type target = child;
368  if (child+1 < mark && compare(data[child], data[child+1]))
369  ++target;
370  // target now has the higher priority child
371  if (compare(data[target], data[data.size()-1])) break;
372  data[cur_pos] = data[target];
373  cur_pos = target;
374  child = (cur_pos<<1)+1;
375  }
376  data[cur_pos] = data[data.size()-1];
377  data.pop_back();
378  if (mark > data.size()) mark = data.size();
379  }
380 };
