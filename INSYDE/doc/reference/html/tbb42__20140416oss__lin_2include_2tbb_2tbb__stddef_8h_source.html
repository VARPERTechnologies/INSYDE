406 template<typename argument_integer_type, typename divisor_integer_type>
407 inline argument_integer_type modulo_power_of_two(argument_integer_type arg, divisor_integer_type divisor) {
408  // Divisor is assumed to be a power of two (which is valid for current uses).
409  __TBB_ASSERT( is_power_of_two(divisor), "Divisor should be a power of two" );
410  return (arg & (divisor - 1));

415 // i.e. for strictly positive i and j, with j a power of 2,
416 // determines whether i==j<<k for some nonnegative k (so i==j yields true).
417 template<typename argument_integer_type, typename divisor_integer_type>
418 inline bool is_power_of_two_factor(argument_integer_type arg, divisor_integer_type divisor) {
419  // Divisor is assumed to be a power of two (which is valid for current uses).
420  __TBB_ASSERT( is_power_of_two(divisor), "Divisor should be a power of two" );
421  return 0 == (arg & (arg - divisor));
422 }
