52  static const intptr_t scheduled = 0x1;
53 
54  F my_func;
55  intptr_t my_state;
56 
57  void mark_scheduled () {
58  // The check here is intentionally lax to avoid the impact of interlocked operation
59  if ( my_state & scheduled )
60  internal::throw_exception( internal::eid_invalid_multiple_scheduling );
61  my_state |= scheduled;
62  }
63 public:
64  task_handle( const F& f ) : my_func(f), my_state(0) {}
65 
66  void operator() () const { my_func(); }
67 };

88 class task_group_base : internal::no_copy {
89 protected:
90  empty_task* my_root;
91  task_group_context my_context;
92 
93  task& owner () { return *my_root; }
94 
95  template<typename F>
96  task_group_status internal_run_and_wait( F& f ) {
97  __TBB_TRY {
98  if ( !my_context.is_group_execution_cancelled() )
99  f();
100  } __TBB_CATCH( ... ) {
101  my_context.register_pending_exception();
102  }
103  return wait();
104  }
105 
106  template<typename F, typename Task>
107  void internal_run( F& f ) {
108  owner().spawn( *new( owner().allocate_additional_child_of(*my_root) ) Task(f) );
109  }
110 
111 public:
112  task_group_base( uintptr_t traits = 0 )
113  : my_context(task_group_context::bound, task_group_context::default_traits | traits)
114  {
115  my_root = new( task::allocate_root(my_context) ) empty_task;
116  my_root->set_ref_count(1);
117  }
118 
119  ~task_group_base() __TBB_NOEXCEPT(false) {
120  if( my_root->ref_count() > 1 ) {
121  bool stack_unwinding_in_progress = std::uncaught_exception();
122  // Always attempt to do proper cleanup to avoid inevitable memory corruption
123  // in case of missing wait (for the sake of better testability & debuggability)
124  if ( !is_canceling() )
125  cancel();
126  __TBB_TRY {
127  my_root->wait_for_all();
128  } __TBB_CATCH (...) {
129  task::destroy(*my_root);
130  __TBB_RETHROW();
131  }
132  task::destroy(*my_root);
133  if ( !stack_unwinding_in_progress )
134  internal::throw_exception( internal::eid_missing_wait );
135  }
136  else {
137  task::destroy(*my_root);
138  }
139  }
140 
141  template<typename F>
142  void run( task_handle<F>& h ) {
143  internal_run< task_handle<F>, internal::task_handle_task<F> >( h );
144  }
145 
146  task_group_status wait() {
147  __TBB_TRY {
148  my_root->wait_for_all();
149  } __TBB_CATCH( ... ) {
150  my_context.reset();
151  __TBB_RETHROW();
152  }
153  if ( my_context.is_group_execution_cancelled() ) {
154  my_context.reset();
155  return canceled;
156  }
157  return complete;
158  }
159 
160  bool is_canceling() {
161  return my_context.is_group_execution_cancelled();
162  }
163 
164  void cancel() {
165  my_context.cancel_group_execution();
166  }
167 }; // class task_group_base
