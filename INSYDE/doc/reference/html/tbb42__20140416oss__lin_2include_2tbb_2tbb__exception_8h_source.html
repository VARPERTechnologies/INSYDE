273 {

275 
276 public:
277  movable_exception ( const ExceptionData& data_ )
278  : my_exception_data(data_)
279  , my_dynamic(false)
280  , my_exception_name(
281 #if TBB_USE_EXCEPTIONS
282  typeid(self_type).name()
283 #else /* !TBB_USE_EXCEPTIONS */
284  "movable_exception"
285 #endif /* !TBB_USE_EXCEPTIONS */
286  )
287  {}
288 
289  movable_exception ( const movable_exception& src ) throw ()
290  : tbb_exception(src)
291  , my_exception_data(src.my_exception_data)
292  , my_dynamic(false)
293  , my_exception_name(src.my_exception_name)
294  {}
295 
296  ~movable_exception () throw() {}
297 
298  const movable_exception& operator= ( const movable_exception& src ) {
299  if ( this != &src ) {
300  my_exception_data = src.my_exception_data;
301  my_exception_name = src.my_exception_name;
302  }
303  return *this;
304  }
305 
306  ExceptionData& data () throw() { return my_exception_data; }
307 
308  const ExceptionData& data () const throw() { return my_exception_data; }
309 
310  /*override*/ const char* name () const throw() { return my_exception_name; }
311 
312  /*override*/ const char* what () const throw() { return "tbb::movable_exception"; }
313 
314  /*override*/
315  movable_exception* move () throw() {
316  void* e = internal::allocate_via_handler_v3(sizeof(movable_exception));
317  if ( e ) {
318  ::new (e) movable_exception(*this);
319  ((movable_exception*)e)->my_dynamic = true;
320  }
321  return (movable_exception*)e;
322  }
323  /*override*/
324  void destroy () throw() {
325  __TBB_ASSERT ( my_dynamic, "Method destroy can be called only on dynamically allocated movable_exceptions" );
326  if ( my_dynamic ) {
327  this->~movable_exception();
328  internal::deallocate_via_handler_v3(this);
329  }
330  }
331  /*override*/
332  void throw_self () { __TBB_THROW( *this ); }
333 
334 protected:
336  ExceptionData my_exception_data;
337 
338 private:

344  const char* my_exception_name;
345 };
