63 // API to retrieve/update FPU control setting
64 #ifndef __TBB_CPU_CTL_ENV_PRESENT
65 #define __TBB_CPU_CTL_ENV_PRESENT 1
66 namespace tbb {
67 namespace internal {
68 class cpu_ctl_env {
69 private:
70  int mxcsr;
71  short x87cw;
72  static const int MXCSR_CONTROL_MASK = ~0x3f; /* all except last six status bits */
73 public:
74  bool operator!=( const cpu_ctl_env& ctl ) const { return mxcsr != ctl.mxcsr || x87cw != ctl.x87cw; }
75  void get_env() {
76  #if __TBB_ICC_12_0_INL_ASM_FSTCW_BROKEN
77  cpu_ctl_env loc_ctl;
78  __asm__ __volatile__ (
79  "stmxcsr %0\n\t"
80  "fstcw %1"
81  : "=m"(loc_ctl.mxcsr), "=m"(loc_ctl.x87cw)
82  );
83  *this = loc_ctl;
84  #else
85  __asm__ __volatile__ (
86  "stmxcsr %0\n\t"
87  "fstcw %1"
88  : "=m"(mxcsr), "=m"(x87cw)
89  );
90  #endif
91  mxcsr &= MXCSR_CONTROL_MASK;
92  }
93  void set_env() const {
94  __asm__ __volatile__ (
95  "ldmxcsr %0\n\t"
96  "fldcw %1"
97  : : "m"(mxcsr), "m"(x87cw)
98  );
99  }
100 };
101 } // namespace internal
102 } // namespace tbb
103 #endif /* !__TBB_CPU_CTL_ENV_PRESENT */
