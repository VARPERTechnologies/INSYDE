122 class task_arena : public internal::task_arena_base {
123  friend class tbb::internal::task_scheduler_observer_v3;
124  bool my_initialized;
125  void mark_initialized() {
126  __TBB_ASSERT( my_arena, "task_arena initialization is incomplete" );
127 #if __TBB_TASK_GROUP_CONTEXT
128  __TBB_ASSERT( my_context, "task_arena initialization is incomplete" );
129 #endif
130 #if TBB_USE_THREADING_TOOLS
131  // Actual synchronization happens in internal_initialize & internal_attach.
132  // The race on setting my_initialized is benign, but should be hidden from Intel(R) Inspector
133  internal::as_atomic(my_initialized).fetch_and_store<release>(true);
134 #else
135  my_initialized = true;
136 #endif
137  }
138 
139 public:

146  task_arena(int max_concurrency = automatic, unsigned reserved_for_masters = 1)
147  : task_arena_base(max_concurrency, reserved_for_masters)
148  , my_initialized(false)
149  {}

152  task_arena(const task_arena &s) // copy settings but not the reference or instance

158  struct attach {};

165  internal_attach();
166  if( my_arena ) my_initialized = true;
167  }

170  inline void initialize() {
171  if( !my_initialized ) {
172  internal_initialize();
173  mark_initialized();
174  }
175  }

178  inline void initialize(int max_concurrency, unsigned reserved_for_masters = 1) {
179  // TODO: decide if this call must be thread-safe
180  __TBB_ASSERT( !my_arena, "Impossible to modify settings of an already initialized task_arena");
181  if( !my_initialized ) {
182  my_max_concurrency = max_concurrency;
183  my_master_slots = reserved_for_masters;
184  initialize();
185  }
186  }
187 
189  inline void initialize(attach) {
190  // TODO: decide if this call must be thread-safe
191  __TBB_ASSERT( !my_arena, "Impossible to modify settings of an already initialized task_arena");
192  if( !my_initialized ) {
193  internal_attach();
194  if( !my_arena ) internal_initialize();
195  mark_initialized();
196  }
197  }

201  inline void terminate() {
202  if( my_initialized ) {
203  internal_terminate();
204  my_initialized = false;
205  }
206  }
