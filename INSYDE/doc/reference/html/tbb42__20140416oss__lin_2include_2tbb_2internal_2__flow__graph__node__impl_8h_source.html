169  void handle_operations(my_operation *op_list) {
170  my_operation *tmp;
171  while (op_list) {
172  tmp = op_list;
173  op_list = op_list->next;
174  switch (tmp->type) {
175  case reg_pred:
176  my_predecessors.add(*(tmp->r));
177  __TBB_store_with_release(tmp->status, SUCCEEDED);
178  if (!forwarder_busy) {
179  forwarder_busy = true;
180  spawn_forward_task();
181  }
182  break;
183  case rem_pred:
184  my_predecessors.remove(*(tmp->r));
185  __TBB_store_with_release(tmp->status, SUCCEEDED);
186  break;
187  case app_body:
188  __TBB_ASSERT(my_max_concurrency != 0, NULL);
189  --my_concurrency;
190  __TBB_store_with_release(tmp->status, SUCCEEDED);
191  if (my_concurrency<my_max_concurrency) {
192  input_type i;
193  bool item_was_retrieved = false;
194  if ( my_queue )
195  item_was_retrieved = my_queue->pop(i);
196  else
197  item_was_retrieved = my_predecessors.get_item(i);
198  if (item_was_retrieved) {
199  ++my_concurrency;
200  spawn_body_task(i);
201  }
202  }
203  break;
204  case app_body_bypass: {
205  task * new_task = NULL;
206  __TBB_ASSERT(my_max_concurrency != 0, NULL);
207  --my_concurrency;
208  if (my_concurrency<my_max_concurrency) {
209  input_type i;
210  bool item_was_retrieved = false;
211  if ( my_queue )
212  item_was_retrieved = my_queue->pop(i);
213  else
214  item_was_retrieved = my_predecessors.get_item(i);
215  if (item_was_retrieved) {
216  ++my_concurrency;
217  new_task = create_body_task(i);
218  }
219  }
220  tmp->bypass_t = new_task;
221  __TBB_store_with_release(tmp->status, SUCCEEDED);
222  }
223  break;
224  case tryput_bypass: internal_try_put_task(tmp); break;
225  case try_fwd: internal_forward(tmp); break;
226  }
227  }
228  }

231  void internal_try_put_task(my_operation *op) {
232  __TBB_ASSERT(my_max_concurrency != 0, NULL);
233  if (my_concurrency < my_max_concurrency) {
234  ++my_concurrency;
235  task * new_task = create_body_task(*(op->elem));
236  op->bypass_t = new_task;
237  __TBB_store_with_release(op->status, SUCCEEDED);
238  } else if ( my_queue && my_queue->push(*(op->elem)) ) {
239  op->bypass_t = SUCCESSFULLY_ENQUEUED;
240  __TBB_store_with_release(op->status, SUCCEEDED);
241  } else {
242  op->bypass_t = NULL;
243  __TBB_store_with_release(op->status, FAILED);
244  }
245  }

248  void internal_forward(my_operation *op) {
249  op->bypass_t = NULL;
250  if (my_concurrency<my_max_concurrency || !my_max_concurrency) {
251  input_type i;
252  bool item_was_retrieved = false;
253  if ( my_queue )
254  item_was_retrieved = my_queue->pop(i);
255  else
256  item_was_retrieved = my_predecessors.get_item(i);
257  if (item_was_retrieved) {
258  ++my_concurrency;
259  op->bypass_t = create_body_task(i);
260  __TBB_store_with_release(op->status, SUCCEEDED);
261  return;
262  }
263  }
264  __TBB_store_with_release(op->status, FAILED);
265  forwarder_busy = false;
266  }

269  // then decides if more work is available
270  void apply_body( input_type &i ) {
271  task *new_task = apply_body_bypass(i);
272  if(!new_task) return;
273  if(new_task == SUCCESSFULLY_ENQUEUED) return;
274  FLOW_SPAWN(*new_task);
275  return;
276  }
277 
279  // then decides if more work is available
280  task * apply_body_bypass( input_type &i ) {
281  task * new_task = static_cast<ImplType *>(this)->apply_body_impl_bypass(i);
282  if ( my_max_concurrency != 0 ) {
283  my_operation op_data(app_body_bypass); // tries to pop an item or get_item, enqueues another apply_body
284  my_aggregator.execute(&op_data);
285  tbb::task *ttask = op_data.bypass_t;
286  new_task = combine_tasks(new_task, ttask);
287  }
288  return new_task;
289  }

303  task* tp = create_body_task(input);
304  // tp == NULL => g.reset(), which shouldn't occur in concurrent context

311  task *forward_task() {
312  my_operation op_data(try_fwd);
313  task *rval = NULL;
314  do {
315  op_data.status = WAIT;
316  my_aggregator.execute(&op_data);
317  if(op_data.status == SUCCEEDED) {
318  tbb::task *ttask = op_data.bypass_t;
319  rval = combine_tasks(rval, ttask);
320  }
321  } while (op_data.status == SUCCEEDED);
322  return rval;
323  }
