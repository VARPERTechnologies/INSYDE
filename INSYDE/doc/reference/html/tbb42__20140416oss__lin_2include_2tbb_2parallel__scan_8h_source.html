56  template<typename Range, typename Body>
57  class final_sum: public task {
58  public:
59  Body my_body;
60  private:
61  aligned_space<Range,1> my_range;
63  Body* my_stuff_last;
64  public:
65  final_sum( Body& body_ ) :
66  my_body(body_,split())
67  {
68  poison_pointer(my_stuff_last);
69  }
70  ~final_sum() {
71  my_range.begin()->~Range();
72  }
73  void finish_construction( const Range& range_, Body* stuff_last_ ) {
74  new( my_range.begin() ) Range(range_);
75  my_stuff_last = stuff_last_;
76  }
77  private:
78  /*override*/ task* execute() {
79  my_body( *my_range.begin(), final_scan_tag() );
80  if( my_stuff_last )
81  my_stuff_last->assign(my_body);
82  return NULL;
83  }
84  };

88  template<typename Range, typename Body>
89  class sum_node: public task {
90  typedef final_sum<Range,Body> final_sum_type;
91  public:
92  final_sum_type *my_incoming;
93  final_sum_type *my_body;
94  Body *my_stuff_last;
95  private:
96  final_sum_type *my_left_sum;
97  sum_node *my_left;
98  sum_node *my_right;
99  bool my_left_is_final;
100  Range my_range;
101  sum_node( const Range range_, bool left_is_final_ ) :
102  my_left_sum(NULL),
103  my_left(NULL),
104  my_right(NULL),
105  my_left_is_final(left_is_final_),
106  my_range(range_)
107  {
108  // Poison fields that will be set by second pass.
109  poison_pointer(my_body);
110  poison_pointer(my_incoming);
111  }
112  task* create_child( const Range& range_, final_sum_type& f, sum_node* n, final_sum_type* incoming_, Body* stuff_last_ ) {
113  if( !n ) {
114  f.recycle_as_child_of( *this );
115  f.finish_construction( range_, stuff_last_ );
116  return &f;
117  } else {
118  n->my_body = &f;
119  n->my_incoming = incoming_;
120  n->my_stuff_last = stuff_last_;
121  return n;
122  }
123  }
124  /*override*/ task* execute() {
125  if( my_body ) {
126  if( my_incoming )
127  my_left_sum->my_body.reverse_join( my_incoming->my_body );
128  recycle_as_continuation();
129  sum_node& c = *this;
130  task* b = c.create_child(Range(my_range,split()),*my_left_sum,my_right,my_left_sum,my_stuff_last);
131  task* a = my_left_is_final ? NULL : c.create_child(my_range,*my_body,my_left,my_incoming,NULL);
132  set_ref_count( (a!=NULL)+(b!=NULL) );
133  my_body = NULL;
134  if( a ) spawn(*b);
135  else a = b;
136  return a;
137  } else {
138  return NULL;
139  }
140  }

142  friend class start_scan;

145  friend class finish_scan;

150  template<typename Range, typename Body>
151  class finish_scan: public task {
152  typedef sum_node<Range,Body> sum_node_type;
153  typedef final_sum<Range,Body> final_sum_type;
154  final_sum_type** const my_sum;
155  sum_node_type*& my_return_slot;
156  public:
157  final_sum_type* my_right_zombie;
158  sum_node_type& my_result;
159 
160  /*override*/ task* execute() {
161  __TBB_ASSERT( my_result.ref_count()==(my_result.my_left!=NULL)+(my_result.my_right!=NULL), NULL );
162  if( my_result.my_left )
163  my_result.my_left_is_final = false;
164  if( my_right_zombie && my_sum )
165  ((*my_sum)->my_body).reverse_join(my_result.my_left_sum->my_body);
166  __TBB_ASSERT( !my_return_slot, NULL );
167  if( my_right_zombie || my_result.my_right ) {
168  my_return_slot = &my_result;
169  } else {
170  destroy( my_result );
171  }
172  if( my_right_zombie && !my_sum && !my_result.my_right ) {
173  destroy(*my_right_zombie);
174  my_right_zombie = NULL;
175  }
176  return NULL;
177  }
178 
179  finish_scan( sum_node_type*& return_slot_, final_sum_type** sum_, sum_node_type& result_ ) :
180  my_sum(sum_),
181  my_return_slot(return_slot_),
182  my_right_zombie(NULL),
183  my_result(result_)
184  {
185  __TBB_ASSERT( !my_return_slot, NULL );
186  }

233  static void run( const Range& range_, Body& body_, const Partitioner& partitioner_ ) {
234  if( !range_.empty() ) {
235  typedef internal::start_scan<Range,Body,Partitioner> start_pass1_type;
236  internal::sum_node<Range,Body>* root = NULL;
237  typedef internal::final_sum<Range,Body> final_sum_type;
238  final_sum_type* temp_body = new(task::allocate_root()) final_sum_type( body_ );
239  start_pass1_type& pass1 = *new(task::allocate_root()) start_pass1_type(
240  /*my_return_slot=*/root,
241  range_,
242  *temp_body,
243  partitioner_ );
244  task::spawn_root_and_wait( pass1 );
245  if( root ) {
246  root->my_body = temp_body;
247  root->my_incoming = NULL;
248  root->my_stuff_last = &body_;
249  task::spawn_root_and_wait( *root );
250  } else {
251  body_.assign(temp_body->my_body);
252  temp_body->finish_construction( range_, NULL );
253  temp_body->destroy(*temp_body);
254  }
255  }
256  }

259  template<typename Range, typename Body, typename Partitioner>
260  task* start_scan<Range,Body,Partitioner>::execute() {
261  typedef internal::finish_scan<Range,Body> finish_pass1_type;
262  finish_pass1_type* p = my_parent_sum ? static_cast<finish_pass1_type*>( parent() ) : NULL;
263  // Inspecting p->result.left_sum would ordinarily be a race condition.
264  // But we inspect it only if we are not a stolen task, in which case we
265  // know that task assigning to p->result.left_sum has completed.
266  bool treat_as_stolen = my_is_right_child && (is_stolen_task() || my_body!=p->my_result.my_left_sum);
267  if( treat_as_stolen ) {
268  // Invocation is for right child that has been really stolen or needs to be virtually stolen
269  p->my_right_zombie = my_body = new( allocate_root() ) final_sum_type(my_body->my_body);
270  my_is_final = false;
271  }
272  task* next_task = NULL;
273  if( (my_is_right_child && !treat_as_stolen) || !my_range.is_divisible() || my_partition.should_execute_range(*this) ) {
274  if( my_is_final )
275  (my_body->my_body)( my_range, final_scan_tag() );
276  else if( my_sum )
277  (my_body->my_body)( my_range, pre_scan_tag() );
278  if( my_sum )
279  *my_sum = my_body;
280  __TBB_ASSERT( !*my_return_slot, NULL );
281  } else {
282  sum_node_type* result;
283  if( my_parent_sum )
284  result = new(allocate_additional_child_of(*my_parent_sum)) sum_node_type(my_range,/*my_left_is_final=*/my_is_final);
285  else
286  result = new(task::allocate_root()) sum_node_type(my_range,/*my_left_is_final=*/my_is_final);
287  finish_pass1_type& c = *new( allocate_continuation()) finish_pass1_type(*my_return_slot,my_sum,*result);
288  // Split off right child
289  start_scan& b = *new( c.allocate_child() ) start_scan( /*my_return_slot=*/result->my_right, *this, result );
290  b.my_is_right_child = true;
291  // Left child is recycling of *this. Must recycle this before spawning b,
292  // otherwise b might complete and decrement c.ref_count() to zero, which
293  // would cause c.execute() to run prematurely.
294  recycle_as_child_of(c);
295  c.set_ref_count(2);
296  c.spawn(b);
297  my_sum = &result->my_left_sum;
298  my_return_slot = &result->my_left;
299  my_is_right_child = false;
300  next_task = this;
301  my_parent_sum = result;
302  __TBB_ASSERT( !*my_return_slot, NULL );
303  }
304  return next_task;
305  }
