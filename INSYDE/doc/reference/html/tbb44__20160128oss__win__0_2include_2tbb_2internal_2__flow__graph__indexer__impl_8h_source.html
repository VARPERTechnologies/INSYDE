32  // Output of the indexer_node is a tbb::flow::tagged_msg, and will be of
33  // the form tagged_msg<tag, result>
34  // where the value of tag will indicate which result was put to the

37  template<typename IndexerNodeBaseType, typename T, size_t K>
38  task* do_try_put(const T &v, void *p) {
39  typename IndexerNodeBaseType::output_type o(K, v);
40  return reinterpret_cast<IndexerNodeBaseType *>(p)->try_put_task(&o);
41  }

169  template<typename InputTuple, typename OutputType, typename StructTypes>
170  class indexer_node_base : public graph_node, public indexer_node_FE<InputTuple, OutputType,StructTypes>,
171  public sender<OutputType> {
172  protected:
173  using graph_node::my_graph;
174  public:
175  static const size_t N = tbb::flow::tuple_size<InputTuple>::value;
176  typedef OutputType output_type;
177  typedef StructTypes tuple_types;
178  typedef receiver<output_type> successor_type;
179  typedef indexer_node_FE<InputTuple, output_type,StructTypes> input_ports_type;
180 #if TBB_PREVIEW_FLOW_GRAPH_FEATURES
181  typedef typename sender<output_type>::built_successors_type built_successors_type;
182  typedef typename sender<output_type>::successor_list_type successor_list_type;
183 #endif
184 
185  private:
186  // ----------- Aggregator ------------
187  enum op_type { reg_succ, rem_succ, try__put_task
188 #if TBB_PREVIEW_FLOW_GRAPH_FEATURES
189  , add_blt_succ, del_blt_succ,
190  blt_succ_cnt, blt_succ_cpy
191 #endif
192  };
193  enum op_stat {WAIT=0, SUCCEEDED, FAILED};
194  typedef indexer_node_base<InputTuple,output_type,StructTypes> class_type;
195 
196  class indexer_node_base_operation : public aggregated_operation<indexer_node_base_operation> {
197  public:
198  char type;
199  union {
200  output_type const *my_arg;
201  successor_type *my_succ;
202  task *bypass_t;
203 #if TBB_PREVIEW_FLOW_GRAPH_FEATURES
204  size_t cnt_val;
205  successor_list_type *succv;
206 #endif
207  };
208  indexer_node_base_operation(const output_type* e, op_type t) :
209  type(char(t)), my_arg(e) {}
210  indexer_node_base_operation(const successor_type &s, op_type t) : type(char(t)),
211  my_succ(const_cast<successor_type *>(&s)) {}
212  indexer_node_base_operation(op_type t) : type(char(t)) {}
213  };
214 
215  typedef internal::aggregating_functor<class_type, indexer_node_base_operation> handler_type;
216  friend class internal::aggregating_functor<class_type, indexer_node_base_operation>;
217  aggregator<handler_type, indexer_node_base_operation> my_aggregator;
218 
219  void handle_operations(indexer_node_base_operation* op_list) {
220  indexer_node_base_operation *current;
221  while(op_list) {
222  current = op_list;
223  op_list = op_list->next;
224  switch(current->type) {
225 
226  case reg_succ:
227  my_successors.register_successor(*(current->my_succ));
228  __TBB_store_with_release(current->status, SUCCEEDED);
229  break;
230 
231  case rem_succ:
232  my_successors.remove_successor(*(current->my_succ));
233  __TBB_store_with_release(current->status, SUCCEEDED);
234  break;
235  case try__put_task: {
236  current->bypass_t = my_successors.try_put_task(*(current->my_arg));
237  __TBB_store_with_release(current->status, SUCCEEDED); // return of try_put_task actual return value
238  }
239  break;
240 #if TBB_PREVIEW_FLOW_GRAPH_FEATURES
241  case add_blt_succ:
242  my_successors.internal_add_built_successor(*(current->my_succ));
243  __TBB_store_with_release(current->status, SUCCEEDED);
244  break;
245  case del_blt_succ:
246  my_successors.internal_delete_built_successor(*(current->my_succ));
247  __TBB_store_with_release(current->status, SUCCEEDED);
248  break;
249  case blt_succ_cnt:
250  current->cnt_val = my_successors.successor_count();
251  __TBB_store_with_release(current->status, SUCCEEDED);
252  break;
253  case blt_succ_cpy:
254  my_successors.copy_successors(*(current->succv));
255  __TBB_store_with_release(current->status, SUCCEEDED);
256  break;
257 #endif /* TBB_PREVIEW_FLOW_GRAPH_FEATURES */
258  }
259  }
260  }
261  // ---------- end aggregator -----------
262  public:
263  indexer_node_base(graph& g) : graph_node(g), input_ports_type() {
264  indexer_helper<StructTypes,N>::set_indexer_node_pointer(this->my_inputs, this);
265  my_successors.set_owner(this);
266  my_aggregator.initialize_handler(handler_type(this));
267  }
268 
269  indexer_node_base(const indexer_node_base& other) : graph_node(other.my_graph), input_ports_type(), sender<output_type>() {
270  indexer_helper<StructTypes,N>::set_indexer_node_pointer(this->my_inputs, this);
271  my_successors.set_owner(this);
272  my_aggregator.initialize_handler(handler_type(this));
273  }
274 
275  bool register_successor(successor_type &r) {
276  indexer_node_base_operation op_data(r, reg_succ);
277  my_aggregator.execute(&op_data);
278  return op_data.status == SUCCEEDED;
279  }
280 
281  bool remove_successor( successor_type &r) {
282  indexer_node_base_operation op_data(r, rem_succ);
283  my_aggregator.execute(&op_data);
284  return op_data.status == SUCCEEDED;
285  }
286 
287  task * try_put_task(output_type const *v) {
288  indexer_node_base_operation op_data(v, try__put_task);
289  my_aggregator.execute(&op_data);
290  return op_data.bypass_t;
291  }
292 
293 #if TBB_PREVIEW_FLOW_GRAPH_FEATURES
294 
295  built_successors_type &built_successors() { return my_successors.built_successors(); }
296 
297  void internal_add_built_successor( successor_type &r) {
298  indexer_node_base_operation op_data(r, add_blt_succ);
299  my_aggregator.execute(&op_data);
300  }
301 
302  void internal_delete_built_successor( successor_type &r) {
303  indexer_node_base_operation op_data(r, del_blt_succ);
304  my_aggregator.execute(&op_data);
305  }
306 
307  size_t successor_count() {
308  indexer_node_base_operation op_data(blt_succ_cnt);
309  my_aggregator.execute(&op_data);
310  return op_data.cnt_val;
311  }
312 
313  void copy_successors( successor_list_type &v) {
314  indexer_node_base_operation op_data(blt_succ_cpy);
315  op_data.succv = &v;
316  my_aggregator.execute(&op_data);
317  }
318  void extract() {
319  my_successors.built_successors().sender_extract(*this);
320  indexer_helper<StructTypes,N>::extract(this->my_inputs);
321  }
322 #endif /* TBB_PREVIEW_FLOW_GRAPH_FEATURES */
323  protected:
324  /*override*/void reset_node(reset_flags f) {
325  if(f & rf_clear_edges) {
326  my_successors.clear();
327  indexer_helper<StructTypes,N>::reset_inputs(this->my_inputs,f);
328  }
329  }
330 
331  private:
332  broadcast_cache<output_type, null_rw_mutex> my_successors;
333  }; //indexer_node_base
