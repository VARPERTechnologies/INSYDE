231 template<typename T>
232 class scalable_allocator {
233 public:
234  typedef typename internal::allocator_type<T>::value_type value_type;
235  typedef value_type* pointer;
236  typedef const value_type* const_pointer;
237  typedef value_type& reference;
238  typedef const value_type& const_reference;
239  typedef size_t size_type;
240  typedef ptrdiff_t difference_type;
241  template<class U> struct rebind {
242  typedef scalable_allocator<U> other;
243  };
244 
245  scalable_allocator() throw() {}
246  scalable_allocator( const scalable_allocator& ) throw() {}
247  template<typename U> scalable_allocator(const scalable_allocator<U>&) throw() {}
248 
249  pointer address(reference x) const {return &x;}
250  const_pointer address(const_reference x) const {return &x;}
251 
253  pointer allocate( size_type n, const void* /*hint*/ =0 ) {
254  return static_cast<pointer>( scalable_malloc( n * sizeof(value_type) ) );
255  }
256 
258  void deallocate( pointer p, size_type ) {
259  scalable_free( p );
260  }
261 
263  size_type max_size() const throw() {
264  size_type absolutemax = static_cast<size_type>(-1) / sizeof (value_type);
265  return (absolutemax > 0 ? absolutemax : 1);
266  }

278  void destroy( pointer p ) {p->~value_type();}
