71 #define __TBB_MACHINE_DEFINE_ATOMICS(S,T,U,A,C) \
72 static inline T __TBB_machine_cmpswp##S ( volatile void * ptr, U value, U comparand ) { \
73  T result; \
74  volatile T *p = (T *)ptr; \
75  __asm \
76  { \
77  __asm mov edx, p \
78  __asm mov C , value \
79  __asm mov A , comparand \
80  __asm lock cmpxchg [edx], C \
81  __asm mov result, A \
82  } \
83  return result; \
84 } \
85 \
86 static inline T __TBB_machine_fetchadd##S ( volatile void * ptr, U addend ) { \
87  T result; \
88  volatile T *p = (T *)ptr; \
89  __asm \
90  { \
91  __asm mov edx, p \
92  __asm mov A, addend \
93  __asm lock xadd [edx], A \
94  __asm mov result, A \
95  } \
96  return result; \
97 }\
98 \
99 static inline T __TBB_machine_fetchstore##S ( volatile void * ptr, U value ) { \
100  T result; \
101  volatile T *p = (T *)ptr; \
102  __asm \
103  { \
104  __asm mov edx, p \
105  __asm mov A, value \
106  __asm lock xchg [edx], A \
107  __asm mov result, A \
108  } \
109  return result; \
110 }

120  __asm
121  {
