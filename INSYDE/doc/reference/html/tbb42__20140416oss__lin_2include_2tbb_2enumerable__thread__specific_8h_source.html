71  struct array {
72  array* next;
73  size_t lg_size;
74  slot& at( size_t k ) {
75  return ((slot*)(void*)(this+1))[k];
76  }
77  size_t size() const {return (size_t)1<<lg_size;}
78  size_t mask() const {return size()-1;}
79  size_t start( size_t h ) const {

83  struct slot {
84  key_type key;
85  void* ptr;
86  bool empty() const {return !key;}
87  bool match( key_type k ) const {return key==k;}
88  bool claim( key_type k ) {
89  __TBB_ASSERT(sizeof(tbb::atomic<key_type>)==sizeof(key_type), NULL);
90  return tbb::internal::punned_cast<tbb::atomic<key_type>*>(&key)->compare_and_swap(k,0)==0;
91  }
92  };

97  static key_type key_of_current_thread() {
98  tbb::tbb_thread::id id = tbb::this_tbb_thread::get_id();
99  key_type k;
100  memcpy( &k, &id, sizeof(k) );

107  atomic<array*> my_root;
108  atomic<size_t> my_count;
109  virtual void* create_local() = 0;
110  virtual void* create_array(size_t _size) = 0; // _size in bytes
111  virtual void free_array(void* ptr, size_t _size) = 0; // _size in bytes
112  array* allocate( size_t lg_size ) {
113  size_t n = 1<<lg_size;
114  array* a = static_cast<array*>(create_array( sizeof(array)+n*sizeof(slot) ));
115  a->lg_size = lg_size;
116  std::memset( a+1, 0, n*sizeof(slot) );
117  return a;
118  }
119  void free(array* a) {
120  size_t n = 1<<(a->lg_size);
121  free_array( (void *)a, size_t(sizeof(array)+n*sizeof(slot)) );
122  }
123  static size_t hash( key_type k ) {
124  // Multiplicative hashing. Client should use *upper* bits.
125  // casts required for Mac gcc4.* compiler
126  return uintptr_t(k)*tbb::internal::select_size_t_constant<0x9E3779B9,0x9E3779B97F4A7C15ULL>::value;
127  }

129  ets_base() {my_root=NULL; my_count=0;}
130  virtual ~ets_base(); // g++ complains if this is not virtual...
131  void* table_lookup( bool& exists );
132  void table_clear();
133  // table_find is used in copying ETS, so is not used in concurrent context. So
134  // we don't need itt annotations for it.
135  slot& table_find( key_type k ) {
136  size_t h = hash(k);
137  array* r = my_root;
138  size_t mask = r->mask();
139  for(size_t i = r->start(h);;i=(i+1)&mask) {
140  slot& s = r->at(i);
141  if( s.empty() || s.match(k) )
142  return s;
143  }
144  }
145  void table_reserve_for_copy( const ets_base& other ) {

147  __TBB_ASSERT(!my_count,NULL);
148  if( other.my_root ) {
149  array* a = allocate(other.my_root->lg_size);
150  a->next = NULL;
151  my_root = a;
152  my_count = other.my_count;
153  }

171  template<ets_key_usage_type ETS_key_type>
172  void* ets_base<ETS_key_type>::table_lookup( bool& exists ) {
173  const key_type k = key_of_current_thread();
174 
175  __TBB_ASSERT(k!=0,NULL);
176  void* found;
177  size_t h = hash(k);
178  for( array* r=my_root; r; r=r->next ) {
179  call_itt_notify(acquired,r);
180  size_t mask=r->mask();
181  for(size_t i = r->start(h); ;i=(i+1)&mask) {
182  slot& s = r->at(i);
183  if( s.empty() ) break;
184  if( s.match(k) ) {
185  if( r==my_root ) {
186  // Success at top level
187  exists = true;
188  return s.ptr;
189  } else {
190  // Success at some other level. Need to insert at top level.
191  exists = true;
192  found = s.ptr;
193  goto insert;
194  }
195  }
196  }
197  }
198  // Key does not yet exist. The density of slots in the table does not exceed 0.5,
199  // for if this will occur a new table is allocated with double the current table
200  // size, which is swapped in as the new root table. So an empty slot is guaranteed.
201  exists = false;
202  found = create_local();
203  {
204  size_t c = ++my_count;
205  array* r = my_root;
206  call_itt_notify(acquired,r);
207  if( !r || c>r->size()/2 ) {
208  size_t s = r ? r->lg_size : 2;
209  while( c>size_t(1)<<(s-1) ) ++s;
210  array* a = allocate(s);
211  for(;;) {
212  a->next = r;
213  call_itt_notify(releasing,a);
214  array* new_r = my_root.compare_and_swap(a,r);
215  if( new_r==r ) break;
216  call_itt_notify(acquired, new_r);
217  if( new_r->lg_size>=s ) {
218  // Another thread inserted an equal or bigger array, so our array is superfluous.
219  free(a);
220  break;
221  }
222  r = new_r;
223  }
224  }
225  }
226  insert:
227  // Whether a slot has been found in an older table, or if it has been inserted at this level,
228  // it has already been accounted for in the total. Guaranteed to be room for it, and it is
229  // not present, so search for empty slot and use it.
230  array* ir = my_root;
231  call_itt_notify(acquired, ir);
232  size_t mask = ir->mask();
233  for(size_t i = ir->start(h);;i=(i+1)&mask) {
234  slot& s = ir->at(i);
235  if( s.empty() ) {
236  if( s.claim(k) ) {
237  s.ptr = found;
238  return found;
239  }
240  }
241  }
242  }

522  break;
