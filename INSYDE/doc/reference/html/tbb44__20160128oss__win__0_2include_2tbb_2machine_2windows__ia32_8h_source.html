48 #define __TBB_MACHINE_DEFINE_ATOMICS(S,T,U,A,C) \
49 static inline T __TBB_machine_cmpswp##S ( volatile void * ptr, U value, U comparand ) { \
50  T result; \
51  volatile T *p = (T *)ptr; \
52  __asm \
53  { \
54  __asm mov edx, p \
55  __asm mov C , value \
56  __asm mov A , comparand \
57  __asm lock cmpxchg [edx], C \
58  __asm mov result, A \
59  } \
60  return result; \
61 } \
62 \
63 static inline T __TBB_machine_fetchadd##S ( volatile void * ptr, U addend ) { \
64  T result; \
65  volatile T *p = (T *)ptr; \
66  __asm \
67  { \
68  __asm mov edx, p \
69  __asm mov A, addend \
70  __asm lock xadd [edx], A \
71  __asm mov result, A \
72  } \
73  return result; \
74 }\
75 \
76 static inline T __TBB_machine_fetchstore##S ( volatile void * ptr, U value ) { \
77  T result; \
78  volatile T *p = (T *)ptr; \
79  __asm \
80  { \
81  __asm mov edx, p \
82  __asm mov A, value \
83  __asm lock xchg [edx], A \
84  __asm mov result, A \
85  } \
86  return result; \
87 }
