38  template <typename T, typename A=cache_aligned_allocator<T> >
39  class item_buffer {
40  public:
41  typedef T input_type;
42  typedef T output_type;
43  protected:
44  typedef size_t size_type;
45  typedef std::pair< T, bool > item_type;
46  typedef typename A::template rebind<item_type>::other allocator_type;
47 
48  item_type *my_array;
49  size_type my_array_size;
50  static const size_type initial_buffer_size = 4;
51  size_type my_head;
52  size_type my_tail;
53 
54  bool buffer_empty() { return my_head == my_tail; }
55 
56  item_type &item(size_type i) { return my_array[i & (my_array_size - 1) ]; } // may not be marked valid
57 
58  bool item_valid(size_type i) { return item(i).second; }
59 
60  void fetch_front(T &v) { __TBB_ASSERT(item_valid(my_head), "front not valid"); v = item(my_head).first; }
61  void fetch_back(T &v) { __TBB_ASSERT(item_valid(my_tail-1), "back not valid"); v = item(my_tail-1).first; }
62 
63  void invalidate(size_type i) { __TBB_ASSERT(item_valid(i), "Item not valid"); item(i).second = false; }
64  void validate(size_type i) { __TBB_ASSERT(!item_valid(i), "Item already valid"); item(i).second = true; }
65 
66  void invalidate_front() { invalidate(my_head); }
67  void validate_front() { validate(my_head); }
68  void invalidate_back() { invalidate(my_tail-1); }
69 
70  size_type size() { return my_tail - my_head; }
71  size_type capacity() { return my_array_size; }
72  bool buffer_full() { return size() == capacity(); }
73 
75  void grow_my_array( size_t minimum_size ) {
76  size_type old_size = my_array_size;
77  size_type new_size = old_size ? 2*old_size : initial_buffer_size;
78  while( new_size<minimum_size )
79  new_size*=2;
80 
81  item_type* new_array = allocator_type().allocate(new_size);
82  item_type* old_array = my_array;
83 
84  for( size_type i=0; i<new_size; ++i ) {
85  new (&(new_array[i].first)) input_type;
86  new_array[i].second = false;
87  }
88 
89  size_t t=my_head;
90  for( size_type i=0; i<old_size; ++i, ++t )
91  new_array[t&(new_size-1)] = old_array[t&(old_size-1)];
92  my_array = new_array;
93  my_array_size = new_size;
94  if( old_array ) {
95  for( size_type i=0; i<old_size; ++i, ++t )
96  old_array[i].first.~input_type();
97  allocator_type().deallocate(old_array,old_size);
98  }
99  }
100 
101  bool push_back(T &v) {
102  if(buffer_full()) {
103  grow_my_array(size() + 1);
104  }
105  item(my_tail) = std::make_pair( v, true );
106  ++my_tail;
107  return true;
108  }
109 
110  bool pop_back(T &v) {
111  if (!item_valid(my_tail-1)) {
112  return false;
113  }
114  fetch_back(v);
115  invalidate_back();
116  --my_tail;
117  return true;
118  }
119 
120  bool pop_front(T &v) {
121  if(!item_valid(my_head)) {
122  return false;
123  }
124  fetch_front(v);
125  invalidate_front();
126  ++my_head;
127  return true;
128  }
129 
130  void clean_up_buffer() {
131  if (my_array) {
132  for( size_type i=0; i<my_array_size; ++i ) {
133  my_array[i].first.~input_type();
134  }
135  allocator_type().deallocate(my_array,my_array_size);
136  }
137  my_array = NULL;
138  my_head = my_tail = my_array_size = 0;
139  }
140 
141  public:
143  item_buffer( ) : my_array(NULL), my_array_size(0),
144  my_head(0), my_tail(0) {
145  grow_my_array(initial_buffer_size);
146  }
147 
148  ~item_buffer() {
149  clean_up_buffer();
150  }
151 
152  void reset() { clean_up_buffer(); grow_my_array(initial_buffer_size); }
153 
154  };

159  template<typename T, typename A=cache_aligned_allocator<T> >
160  class reservable_item_buffer : public item_buffer<T, A> {
161  protected:
162  using item_buffer<T, A>::buffer_empty;
163  using item_buffer<T, A>::fetch_front;
164  using item_buffer<T, A>::invalidate_front;
165  using item_buffer<T, A>::validate_front;
166  using item_buffer<T, A>::item_valid;
167  using item_buffer<T, A>::my_head;
168 
169  public:
170  reservable_item_buffer() : item_buffer<T, A>(), my_reserved(false) {}
171  void reset() {my_reserved = false; item_buffer<T,A>::reset(); }
172  protected:
173 
174  bool reserve_front(T &v) {
175  if(my_reserved || !item_valid(my_head)) return false;
176  my_reserved = true;
177  // reserving the head
178  fetch_front(v);
179  // invalidate the head, but don't commit until consume is called
180  invalidate_front();
181  return true;
182  }
183 
184  void consume_front() {
185  __TBB_ASSERT(my_reserved, "Attempt to consume a non-reserved item");
186  ++my_head;
187  my_reserved = false;
188  }
189 
190  void release_front() {
191  __TBB_ASSERT(my_reserved, "Attempt to release a non-reserved item");
192  validate_front();
193  my_reserved = false;
194  }
195 
196  bool my_reserved;
197  };
