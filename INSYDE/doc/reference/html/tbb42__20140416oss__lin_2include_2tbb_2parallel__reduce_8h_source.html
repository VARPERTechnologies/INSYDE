56  template<typename Body>
57  class finish_reduce: public flag_task {
59  bool has_right_zombie;
60  const reduction_context my_context;
61  Body* my_body;
62  aligned_space<Body,1> zombie_space;
63  finish_reduce( reduction_context context_ ) :
64  has_right_zombie(false), // TODO: substitute by flag_task::child_stolen?
65  my_context(context_),
66  my_body(NULL)
67  {
68  }
69  ~finish_reduce() {
70  if( has_right_zombie )
71  zombie_space.begin()->~Body();
72  }
73  task* execute() {
74  if( has_right_zombie ) {
75  // Right child was stolen.
76  Body* s = zombie_space.begin();
77  my_body->join( *s );
78  // Body::join() won't be called if canceled. Defer destruction to destructor
79  }
80  if( my_context==left_child )
81  itt_store_word_with_release( static_cast<finish_reduce*>(parent())->my_body, my_body );
82  return NULL;
83  }
84  template<typename Range,typename Body_, typename Partitioner>
85  friend class start_reduce;
86  };

182  inline void allocate_sibling(task* start_reduce_task, task *tasks[], size_t start_bytes, size_t finish_bytes) {
183  tasks[0] = &start_reduce_task->allocate_continuation().allocate(finish_bytes);

189  template<typename Range, typename Body, typename Partitioner>
190  task* start_reduce<Range,Body,Partitioner>::execute() {
191  my_partition.check_being_stolen( *this );
192  if( my_context==right_child ) {
193  finish_type* parent_ptr = static_cast<finish_type*>(parent());
194  if( !itt_load_word_with_acquire(parent_ptr->my_body) ) { // TODO: replace by is_stolen_task() or by parent_ptr->ref_count() == 2???
195  my_body = new( parent_ptr->zombie_space.begin() ) Body(*my_body,split());
196  parent_ptr->has_right_zombie = true;
197  }
198  } else __TBB_ASSERT(my_context==root_task,NULL);// because left leaf spawns right leafs without recycling
199  my_partition.execute(*this, my_range);
200  if( my_context==left_child ) {
201  finish_type* parent_ptr = static_cast<finish_type*>(parent());
202  __TBB_ASSERT(my_body!=parent_ptr->zombie_space.begin(),NULL);
203  itt_store_word_with_release(parent_ptr->my_body, my_body );
204  }
205  return NULL;
206  }

272  template<typename Range, typename Body>
273  task* start_deterministic_reduce<Range,Body>::execute() {
274  if( !my_range.is_divisible() ) {
275  my_body( my_range );
276  return NULL;
277  } else {
278  finish_type& c = *new( allocate_continuation() ) finish_type( my_body );
279  recycle_as_child_of(c);
280  c.set_ref_count(2);
281  start_deterministic_reduce& b = *new( c.allocate_child() ) start_deterministic_reduce( *this, c );
282  task::spawn(b);
283  return this;
284  }
