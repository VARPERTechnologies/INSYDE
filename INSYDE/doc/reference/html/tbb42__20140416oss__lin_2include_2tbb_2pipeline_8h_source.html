97 #else

102  static const unsigned char version_mask = 0x7<<1; // bits 1-3 are for version
103 public:

108  serial_in_order = current_version | filter_is_serial,

356 // large object helper (uses tbb_allocator)
357 template<typename T>
358 class token_helper<T, true> {
359  public:
360  typedef typename tbb::tbb_allocator<T> allocator;
361  typedef T* pointer;
362  typedef T value_type;
363  static pointer create_token(const value_type & source) {
364  pointer output_t = allocator().allocate(1);
365  return new (output_t) T(source);
366  }
367  static value_type & token(pointer & t) { return *t;}
368  static void * cast_to_void_ptr(pointer ref) { return (void *) ref; }
369  static pointer cast_from_void_ptr(void * ref) { return (pointer)ref; }
370  static void destroy_token(pointer token) {
371  allocator().destroy(token);
372  allocator().deallocate(token,1);
373  }
374 };
375

377 template<typename T>
378 class token_helper<T*, false > {
379  public:
380  typedef T* pointer;
381  typedef T* value_type;
382  static pointer create_token(const value_type & source) { return source; }
383  static value_type & token(pointer & t) { return t;}
384  static void * cast_to_void_ptr(pointer ref) { return (void *)ref; }
385  static pointer cast_from_void_ptr(void * ref) { return (pointer)ref; }
386  static void destroy_token( pointer /*token*/) {}

389 // small object specialization (converts void* to the correct type, passes objects directly.)
390 template<typename T>
391 class token_helper<T, false> {
392  typedef union {
393  T actual_value;
394  void * void_overlay;
395  } type_to_void_ptr_map;
396  public:
397  typedef T pointer; // not really a pointer in this case.
398  typedef T value_type;
399  static pointer create_token(const value_type & source) {
400  return source; }
401  static value_type & token(pointer & t) { return t;}
402  static void * cast_to_void_ptr(pointer ref) {
403  type_to_void_ptr_map mymap;
404  mymap.void_overlay = NULL;
405  mymap.actual_value = ref;
406  return mymap.void_overlay;
407  }
408  static pointer cast_from_void_ptr(void * ref) {
409  type_to_void_ptr_map mymap;
410  mymap.void_overlay = ref;
411  return mymap.actual_value;
412  }
413  static void destroy_token( pointer /*token*/) {}
414 };

594 template<typename T, typename U>
595 class filter_t {
596  typedef internal::filter_node filter_node;
597  filter_node* root;
598  filter_t( filter_node* root_ ) : root(root_) {
599  root->add_ref();
600  }
601  friend class internal::pipeline_proxy;
602  template<typename T_, typename U_, typename Body>
603  friend filter_t<T_,U_> make_filter(tbb::filter::mode, const Body& );
604  template<typename T_, typename V_, typename U_>
605  friend filter_t<T_,U_> operator& (const filter_t<T_,V_>& , const filter_t<V_,U_>& );
606 public:
607  filter_t() : root(NULL) {}
608  filter_t( const filter_t<T,U>& rhs ) : root(rhs.root) {
609  if( root ) root->add_ref();
610  }
611  template<typename Body>
612  filter_t( tbb::filter::mode mode, const Body& body ) :
613  root( new internal::filter_node_leaf<T,U,Body>(mode, body) ) {
614  root->add_ref();
615  }
616 
617  void operator=( const filter_t<T,U>& rhs ) {
618  // Order of operations below carefully chosen so that reference counts remain correct
619  // in unlikely event that remove_ref throws exception.
620  filter_node* old = root;
621  root = rhs.root;
622  if( root ) root->add_ref();
623  if( old ) old->remove_ref();
624  }
625  ~filter_t() {
626  if( root ) root->remove_ref();
627  }
628  void clear() {
629  // Like operator= with filter_t() on right side.
630  if( root ) {
631  filter_node* old = root;
632  root = NULL;
633  old->remove_ref();
634  }
635  }
