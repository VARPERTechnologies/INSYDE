29 // tagged buffer that can expand, and can support as many deletions as additions
30 // list-based, with elements of list held in std::vector (for destruction management),
31 // multiplicative hashing (like ets). No synchronization built-in.
32 //

41 template<typename TagType, typename ValueType, size_t NoTagMark>
42 struct buffer_element {
43  TagType t;
44  ValueType v;
45  buffer_element *next;
46  buffer_element() : t(NoTagMark), next(NULL) {}
47 };
48 
49 template
50  <
51  typename TagType,
52  typename ValueType,
53  size_t NoTagMark = 0,
54  typename Allocator=tbb::cache_aligned_allocator< buffer_element<TagType,ValueType,NoTagMark> >
55  >
56 class tagged_buffer {
57 public:
58  static const size_t INITIAL_SIZE = 8; // initial size of the hash pointer table
59  static const TagType NO_TAG = TagType(NoTagMark);
60  typedef ValueType value_type;
61  typedef buffer_element<TagType,ValueType, NO_TAG> element_type;
62  typedef value_type *pointer_type;
63  typedef std::vector<element_type, Allocator> list_array_type;
64  typedef typename Allocator::template rebind<element_type*>::other pointer_array_allocator_type;
65  typedef typename Allocator::template rebind<list_array_type>::other list_array_allocator;
66 private:
67 
68  size_t my_size;
69  size_t nelements;
70  element_type** array;
71  std::vector<element_type, Allocator> *lists;
72  element_type* free_list;
73 
74  size_t mask() { return my_size - 1; }
75 
76  static size_t hash(TagType t) {
77  return uintptr_t(t)*tbb::internal::select_size_t_constant<0x9E3779B9,0x9E3779B97F4A7C15ULL>::value;
78  }
79 
80  void set_up_free_list( element_type **p_free_list, list_array_type *la, size_t sz) {
81  for(size_t i=0; i < sz - 1; ++i ) { // construct free list
82  (*la)[i].next = &((*la)[i+1]);
83  (*la)[i].t = NO_TAG;
84  }
85  (*la)[sz-1].next = NULL;
86  *p_free_list = &((*la)[0]);
87  }
88 
89  void grow_array() {
90  // make the pointer array larger
91  element_type **new_array;
92  element_type **old_array = array;
93  size_t old_size = my_size;
94  my_size *=2;
95  new_array = pointer_array_allocator_type().allocate(my_size);
96  for(size_t i=0; i < my_size; ++i) new_array[i] = NULL;
97  list_array_type *new_list_array = new list_array_type(old_size, element_type(), Allocator());
98  set_up_free_list(&free_list, new_list_array, old_size );
99 
100  for(size_t i=0; i < old_size; ++i) {
101  for( element_type* op = old_array[i]; op; op = op->next) {
102  internal_tagged_insert(new_array, my_size, op->t, op->v);
103  }
104  }
105  pointer_array_allocator_type().deallocate(old_array, old_size);
106 
107  delete lists; // destroy and deallocate instead
108  array = new_array;
109  lists = new_list_array;
110  }
111 
112  void internal_tagged_insert( element_type **ar, size_t sz, TagType t, value_type v) {
113  size_t l_mask = sz-1;
114  size_t h = hash(t) & l_mask;
115  __TBB_ASSERT(free_list, "Error: free list not set up.");
116  element_type* my_elem = free_list; free_list = free_list->next;
117  my_elem->t = t;
118  my_elem->v = v;
119  my_elem->next = ar[h];
120  ar[h] = my_elem;
121  }
122 
123  void internal_initialize_buffer() {
124  array = pointer_array_allocator_type().allocate(my_size);
125  for(size_t i = 0; i < my_size; ++i) array[i] = NULL;
126  lists = new list_array_type(INITIAL_SIZE/2, element_type(), Allocator());
127  set_up_free_list(&free_list, lists, INITIAL_SIZE/2);
128  }
129 
130  void internal_free_buffer() {
131  if(array) {
132  pointer_array_allocator_type().deallocate(array, my_size);
133  array = NULL;
134  }
135  if(lists) {
136  delete lists;
137  lists = NULL;
138  }
139  my_size = INITIAL_SIZE;
140  nelements = 0;
141  }
142 
143 public:
144  tagged_buffer() : my_size(INITIAL_SIZE), nelements(0) {
145  internal_initialize_buffer();
146  }
147 
148  ~tagged_buffer() {
149  internal_free_buffer();
150  }
151 
152  void reset() {
153  internal_free_buffer();
154  internal_initialize_buffer();
155  }

157  bool tagged_insert(TagType t, value_type v) {
158  pointer_type p;
159  if(tagged_find_ref(t, p)) {
160  *p = v; // replace the value
161  return false;
162  }
163  ++nelements;
164  if(nelements*2 > my_size) grow_array();
165  internal_tagged_insert(array, my_size, t, v);
166  return true;
167  }
168 
169  // returns reference to array element.v
170  bool tagged_find_ref(TagType t, pointer_type &v) {
171  size_t i = hash(t) & mask();
172  for(element_type* p = array[i]; p; p = p->next) {
173  if(p->t == t) {
174  v = &(p->v);
175  return true;
176  }
177  }
178  return false;
179  }
180 
181  bool tagged_find( TagType t, value_type &v) {
182  value_type *p;
183  if(tagged_find_ref(t, p)) {
184  v = *p;
185  return true;
186  }
187  else
188  return false;
189  }
190 
191  void tagged_delete(TagType t) {
192  size_t h = hash(t) & mask();
193  element_type* prev = NULL;
194  for(element_type* p = array[h]; p; prev = p, p = p->next) {
195  if(p->t == t) {
196  p->t = NO_TAG;
197  if(prev) prev->next = p->next;
198  else array[h] = p->next;
199  p->next = free_list;
200  free_list = p;
201  --nelements;
202  return;
203  }
204  }
205  __TBB_ASSERT(false, "tag not found for delete");
206  }
207 
208  // search for v in the array; if found {set t, return true} else return false
209  // we use this in join_node_FE to find if a tag's items are all available.
210  bool find_value_tag( TagType &t, value_type v) {
211  for(size_t i= 0; i < my_size / 2; ++i) { // remember the vector is half the size of the hash array
212  if( (*lists)[i].t != NO_TAG && (*lists)[i].v == v) {
213  t = (*lists)[i].t;
214  return true;
215  }
216  }
217  return false;
218  }
219 };
