48 inline static uint8_t __TBB_machine_try_lock_elided( volatile uint8_t* lk )
49 {
50  uint8_t value = 1;
51  __asm__ volatile (".byte " __TBB_STRINGIZE(__TBB_OP_XACQUIRE)"; lock; xchgb %0, %1;"
52  : __TBB_r_out(value), "=m"(*lk) : "0"(value), "m"(*lk) : "memory" );
53  return uint8_t(value^1);
54 }

78 inline static bool __TBB_machine_is_in_transaction()
79 {

107 inline static uint32_t __TBB_machine_begin_transaction()
108 {
109  uint32_t res = ~uint32_t(0); // success value
110  __asm__ volatile ("1: .byte 0xC7; .byte 0xF8;\n" // XBEGIN <abort-offset>
111  " .long 2f-1b-6\n" // 2f-1b == difference in addresses of start
112  // of XBEGIN and the MOVL
113  // 2f - 1b - 6 == that difference minus the size of the
114  // XBEGIN instruction. This is the abort offset to
115  // 2: below.
116  " jmp 3f\n" // success (leave -1 in res)
117  "2: movl %%eax,%0\n" // store failure code in res
118  "3:"
119  :"=r"(res):"0"(res):"memory","%eax");
120  return res;
121 }
