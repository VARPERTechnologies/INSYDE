32 // wrapper for padding mutexes to be alone on a cache line, without requiring they be allocated
33 // from a pool. Because we allow them to be defined anywhere they must be two cache lines in size.
34 
35 
36 namespace tbb {
37 namespace interface7 {
38 namespace internal {
39 
40 static const size_t cache_line_size = 64;
41 
42 // Pad a mutex to occupy a number of full cache lines sufficient to avoid false sharing
43 // with other data; space overhead is up to 2*cache_line_size-1.
44 template<typename Mutex, bool is_rw> class padded_mutex;
45 
46 template<typename Mutex>
47 class padded_mutex<Mutex,false> {
48  typedef long pad_type;
49  pad_type my_pad[((sizeof(Mutex)+cache_line_size-1)/cache_line_size+1)*cache_line_size/sizeof(pad_type)];
50 
51  Mutex *impl() { return (Mutex *)((uintptr_t(this)|(cache_line_size-1))+1);}
52 
53 public:
54  static const bool is_rw_mutex = Mutex::is_rw_mutex;
55  static const bool is_recursive_mutex = Mutex::is_recursive_mutex;
56  static const bool is_fair_mutex = Mutex::is_fair_mutex;
57 
58  padded_mutex() { new(impl()) Mutex(); }
59  ~padded_mutex() { impl()->~Mutex(); }
60 
62  class scoped_lock : tbb::internal::no_copy {
63  typename Mutex::scoped_lock my_scoped_lock;
64  public:
65  scoped_lock() : my_scoped_lock() {}
66  scoped_lock( padded_mutex& m ) : my_scoped_lock(*m.impl()) { }
67  ~scoped_lock() { }
68 
69  void acquire( padded_mutex& m ) { my_scoped_lock.acquire(*m.impl()); }
70  bool try_acquire( padded_mutex& m ) { return my_scoped_lock.try_acquire(*m.impl()); }
71  void release() { my_scoped_lock.release(); }
72  };
73 };
