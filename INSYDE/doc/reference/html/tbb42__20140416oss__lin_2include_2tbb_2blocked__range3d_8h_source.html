105  template <typename Split>
106  void do_split( blocked_range3d& r, Split& split_obj)
107  {
108  if ( my_pages.size()*double(my_rows.grainsize()) < my_rows.size()*double(my_pages.grainsize()) ) {
109  if ( my_rows.size()*double(my_cols.grainsize()) < my_cols.size()*double(my_rows.grainsize()) ) {
110  my_cols.my_begin = col_range_type::do_split(r.my_cols, split_obj);
111  } else {
112  my_rows.my_begin = row_range_type::do_split(r.my_rows, split_obj);
113  }
114  } else {
115  if ( my_pages.size()*double(my_cols.grainsize()) < my_cols.size()*double(my_pages.grainsize()) ) {
116  my_cols.my_begin = col_range_type::do_split(r.my_cols, split_obj);
117  } else {
118  my_pages.my_begin = page_range_type::do_split(r.my_pages, split_obj);
119  }
120  }
121  }
