50  class start_for: public task {

53  typename Partitioner::task_partition_type my_partition;

64  my_range(range),

67  {
68  }

81  my_range(r),
82  my_body(parent_.my_body),
83  my_partition(parent_.my_partition, split())
84  {
85  my_partition.set_affinity(*this);
86  my_partition.align_depth( d );
87  }
88  static void run( const Range& range, const Body& body, Partitioner& partitioner ) {
89  if( !range.empty() ) {
90 #if !__TBB_TASK_GROUP_CONTEXT || TBB_JOIN_OUTER_TASK_GROUP
91  start_for& a = *new(task::allocate_root()) start_for(range,body,partitioner);
92 #else
93  // Bound context prevents exceptions from body to affect nesting or sibling algorithms,
94  // and allows users to handle exceptions safely by wrapping parallel_for in the try-block.
95  task_group_context context;
96  start_for& a = *new(task::allocate_root(context)) start_for(range,body,partitioner);
97 #endif /* __TBB_TASK_GROUP_CONTEXT && !TBB_JOIN_OUTER_TASK_GROUP */
98  task::spawn_root_and_wait(a);
99  }
100  }
101 #if __TBB_TASK_GROUP_CONTEXT
102  static void run( const Range& range, const Body& body, Partitioner& partitioner, task_group_context& context ) {
103  if( !range.empty() ) {
104  start_for& a = *new(task::allocate_root(context)) start_for(range,body,partitioner);
105  task::spawn_root_and_wait(a);
106  }
107  }
108 #endif /* __TBB_TASK_GROUP_CONTEXT */
109  void run_body( Range &r ) { my_body( r ); }

113  void offer_work(typename Partitioner::split_type& split_obj) {
114  spawn( *new( allocate_sibling(static_cast<task*>(this), sizeof(start_for)) ) start_for(*this, split_obj) );
115  }
117  void offer_work(const Range& r, depth_t d = 0) {
118  spawn( *new( allocate_sibling(static_cast<task*>(this), sizeof(start_for)) ) start_for(*this, r, d) );
119  }

147  template<typename Function, typename Index>
148  class parallel_for_body : internal::no_assign {
149  const Function &my_func;
150  const Index my_begin;
151  const Index my_step;
152  public:
153  parallel_for_body( const Function& _func, Index& _begin, Index& _step )
154  : my_func(_func), my_begin(_begin), my_step(_step) {}
155 
156  void operator()( const tbb::blocked_range<Index>& r ) const {
157  // A set of local variables to help the compiler with vectorization of the following loop.
158  Index b = r.begin();
159  Index e = r.end();
160  Index ms = my_step;
161  Index k = my_begin + b*ms;
162 
163 #if __INTEL_COMPILER
164 #pragma ivdep
165 #if __TBB_ASSERT_ON_VECTORIZATION_FAILURE
166 #pragma vector always assert
167 #endif
168 #endif
169  for ( Index i = b; i < e; ++i, k += ms ) {
170  my_func( k );
171  }
172  }
173  };
174 } // namespace internal

252 template <typename Index, typename Function, typename Partitioner>
254 void parallel_for_impl(Index first, Index last, Index step, const Function& f, Partitioner& partitioner) {
255  if (step <= 0 )
256  internal::throw_exception(internal::eid_nonpositive_step); // throws std::invalid_argument
257  else if (last > first) {
258  // Above "else" avoids "potential divide by zero" warning on some platforms
259  Index end = (last - first - Index(1)) / step + Index(1);
260  tbb::blocked_range<Index> range(static_cast<Index>(0), end);
261  internal::parallel_for_body<Function, Index> body(f, first, step);
262  tbb::parallel_for(range, body, partitioner);
263  }
264 }
